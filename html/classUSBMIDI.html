Detailed Description

USBMIDI example

 #include "mbed.h"
 #include "USBMIDI.h"

 USBMIDI midi;

 int main() {            
    while (1) {    
        for(int i=48; i<83; i++) {     // send some messages!
            midi.write(MIDIMessage::NoteOn(i));
            wait(0.25);
            midi.write(MIDIMessage::NoteOff(i));
            wait(0.5);
        }
    }
 }
