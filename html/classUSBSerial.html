USBSerial example

 #include "mbed.h"
 #include "USBSerial2.h"

 //Virtual serial port over USB
 USBSerial serial;

 int main(void) {

    while(1)
    {
        serial.printf("I am a virtual serial port\n");
        wait(1);
    }
 }
