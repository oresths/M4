Detailed Description

USBHID example 

 #include "mbed.h"
 #include "USBHID.h"

 USBHID hid;
 HID_REPORT recv;
 BusOut leds(LED1,LED2,LED3,LED4);

 int main(void) {
    while (1) {
        hid.read(&recv);
        leds = recv.data[0];
    }
 }
